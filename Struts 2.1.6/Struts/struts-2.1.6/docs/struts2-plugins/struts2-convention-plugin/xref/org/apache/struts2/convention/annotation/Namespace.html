21  package org.apache.struts2.convention.annotation;
22  
23  import java.lang.annotation.Retention;
24  import java.lang.annotation.RetentionPolicy;
25  import java.lang.annotation.Target;
26  import java.lang.annotation.ElementType;
27  
28  /***
29   * <!-- START SNIPPET: javadoc -->
30   * <p>
31   * This annotation defines how actions can modify the namespace
32   * that they are added to. This overrides the behavior of the
33   * Convention plugin which by default uses the package names for
34   * namespaces. Since XWork packages are created by the Convention
35   * plugin via the Java packages that the actions exist in, there
36   * is some tricky handling of XWork parent packages and namespaces
37   * of the XWork packages for the Convention plugin discovered
38   * actions so that two actions in the same package can specify
39   * different parents and namespaces without collision.
40   * </p>
41   *
42   * <p>
43   * In order to handle this correctly, the name of the XWork
44   * package that actions are placed into is built using this
45   * format:
46   * </p>
47   *
48   * <pre>
49   * &lt;java-package>#&lt;parent-xwork-package>#&lt;namespace>
50   * </pre>
51   *
52   * <p>
53   * This mechanism will guarantee that two actions in the same
54   * Java package can specify different parent packages (using the
55   * {@link org.apache.struts2.convention.annotation.ParentPackage} annotation)
56   * and namespaces (using this annotation).
57   * </p>
58   *
59   * <p>
60   * The value of a Namespace annotation should specify the portion of
61   * the action URL between the context path and the action name.  For
62   * example:
63   * </p>
64   * <pre>
65   * &#064;Namespace("/careers/job-postings-overview/job-postings")
66   * </pre>
67   *
68   * <p>
69   * This annotation can also be placed inside the special Java file
70   * named <strong>package-info.java</strong>, which allows package
71   * level annotations. If this is used in this manner it changes the
72   * default namespace for all actions within that Java package. The
73   * search order for the namespace of a particular class is therefore:
74   * </p>
75   *
76   * <ol>
77   * <li>If the {@link Action} annotation exists within the action and
78   *  specifies a full URI (i.e. it starts with a / character)</li>
79   * <li>Any Namespace annotations placed on individual action classes</li>
80   * <li>Any Namespace annotations placed in the package-info.java file</li>
81   * <li>The namespace as determined using the Java package name and the
82   *  standard convention based naming.</li>
83   * </ol>
84   * <!-- END SNIPPET: javadoc -->
85   */
86  @Target({ElementType.PACKAGE, ElementType.TYPE})
87  @Retention(value = RetentionPolicy.RUNTIME)
88  public @interface Namespace {
89      /***
90       * @return  The namespace value.
91       */
92      String value();
93  }
