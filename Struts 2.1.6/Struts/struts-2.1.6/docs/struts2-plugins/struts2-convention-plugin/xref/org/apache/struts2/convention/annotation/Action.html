21  package org.apache.struts2.convention.annotation;
22  
23  import java.lang.annotation.ElementType;
24  import java.lang.annotation.Retention;
25  import java.lang.annotation.RetentionPolicy;
26  import java.lang.annotation.Target;
27  
28  /***
29   * <!-- START SNIPPET: javadoc -->
30   * <p>
31   * This annotation can be used to control the URL that maps to
32   * a specific method in an Action class. By default, the method
33   * that is invoked is the execute method of the action and the
34   * URL is based on the package and class names. This annotation
35   * allows developers to change the URL or invoke a different
36   * method. This also allows developers to specify multiple URLs
37   * that will be handled by a single class or a single method.
38   * </p>
39   * <p>
40   * This can also be used via the {@link Actions} annotation
41   * to associate multiple URLs with a single method.
42   * </p>
43   * <p>
44   * Here's an example:
45   * </p>
46   *
47   * <pre>
48   * public class MyAction implements Action {
49   *   {@code @Action("/foo/bar")}
50   *   public String execute() {}
51   * }
52   * </pre>
53   * <!-- END SNIPPET: javadoc -->
54   */
55  @Target({ElementType.METHOD, ElementType.TYPE})
56  @Retention(RetentionPolicy.RUNTIME)
57  public @interface Action {
58      String DEFAULT_VALUE = "DEFAULT_VALUE";
59  
60      /***
61       * Allows actions to specify different URLs rather than the default that is based on the package
62       * and action name. This also allows methods other than execute() to be invoked or multiple URLs
63       * to map to a single class or a single method to handle multiple URLs.
64       *
65       * @return  The action URL.
66       */
67      String value() default DEFAULT_VALUE;
68  
69      /***
70       * Allows action methods to specifically control the results for specific return values. These
71       * results are not used for other method/URL invocations on the action. These are only used for
72       * the URL that this action is associated with.
73       *
74       * @return  The results for the action.
75       */
76      Result[] results() default {};
77  
78      /***
79       * Allows action methods to specify what interceptors must be applied to it.
80       * @return Interceptors to be applied to the action
81       */
82      InterceptorRef[] interceptorRefs() default {};
83  
84      /***
85       * @return  The parameters passed to the action. This is a list of strings that form a name/value
86       *          pair chain since creating a Map for annotations is not possible. An example would be:
87       *          <code>{"key", "value", "key2", "value2"}</code>.
88       */
89      String[] params() default {};
90  
91      /***
92       * @return Maps return codes to exceptions. The "exceptions" interceptor must be applied to the action.
93       */
94      ExceptionMapping[] exceptionMappings() default {};
95  }
