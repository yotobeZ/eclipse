97       *
98       * @param configuration         The XWork configuration that the new package configs and action configs
99       *                              are added to.
100      * @param actionNameBuilder     The action name builder used to convert action class names to action
101      *                              names.
102      * @param resultMapBuilder      The result map builder used to create ResultConfig mappings for each
103      *                              action.
104      * @param interceptorMapBuilder The interceptor map builder used to create InterceptorConfig mappings for each
105      *                              action.
106      * @param objectFactory         The ObjectFactory used to create the actions and such.

243     /***
244      * Builds the action configurations by loading all classes in the packages specified by the
245      * property <b>struts.convention.action.packages</b> and then figuring out which classes implement Action
246      * or have Action in their name. Next, if this class is in a Java package that hasn't been
247      * inspected a new PackageConfig (XWork) is created for that Java package using the Java package
248      * name. This will contain all the ActionConfigs for all the Action classes that are discovered
249      * within that Java package. Next, each class is inspected for the {@link ParentPackage}
250      * annotation which is used to control the parent package for a specific action. Lastly, the
251      * {@link ResultMapBuilder} is used to create ResultConfig instances of the action.
252      */
253     public void buildActionConfigs() {
254         //setup reload class loader based on dev settings
255         initReloadClassLoader();
256         
257         if (!disableActionScanning) {
258             if (actionPackages == null && packageLocators == null) {
259                 throw new ConfigurationException("At least a list of action packages or action package locators " +
260                         "must be given using one of the properties [struts.convention.action.packages] or " +
261                         "[struts.convention.package.locators]");
262             }
263 
264             if (LOG.isTraceEnabled()) {
265                 LOG.trace("Loading action configurations");
266                 if (actionPackages != null) {
267                     LOG.trace("Actions being loaded from action packages " + Arrays.asList(actionPackages));
268                 }
269                 if (packageLocators != null) {
270                     LOG.trace("Actions being loaded using package locators " + Arrays.asList(packageLocators));
271                 }
272                 if (excludePackages != null) {
273                     LOG.trace("Excluding actions from packages " + Arrays.asList(excludePackages));
274                 }
275             }
276 
277             Set<Class> classes = findActions();
278             buildConfiguration(classes);
279         }
280     }

290     @SuppressWarnings("unchecked")
291     protected Set<Class> findActions() {
292         Set<Class> classes = new HashSet<Class>();
293         try {
294             if (actionPackages != null || (packageLocators != null && !disablePackageLocatorsScanning)) {
295                 ClassFinder finder = new ClassFinder(getClassLoaderForFinder(), buildUrlSet().getUrls(), true);
296 
297                 // named packages
298                 if (actionPackages != null) {
299                     for (String packageName : actionPackages) {
300                         Test<ClassFinder.ClassInfo> test = getPackageFinderTest(packageName);
301                         classes.addAll(finder.findClasses(test));
302                     }
303                 }
304 
305                 //package locators
306                 if (packageLocators != null && !disablePackageLocatorsScanning) {
307                     for (String packageLocator : packageLocators) {
308                         Test<ClassFinder.ClassInfo> test = getPackageLocatorTest(packageLocator);
309                         classes.addAll(finder.findClasses(test));
310                     }
311                 }
312             }
313         } catch (Exception ex) {
314             if (LOG.isErrorEnabled())
315                 LOG.error("Unable to scan named packages", ex);
316         }
317 
318         return classes;
319     }

321     private UrlSet buildUrlSet() throws IOException {
322         UrlSet urlSet = new UrlSet(getClassLoader());
323 
324         urlSet = urlSet.exclude(ClassLoader.getSystemClassLoader().getParent());
325         urlSet = urlSet.excludeJavaExtDirs();
326         urlSet = urlSet.excludeJavaEndorsedDirs();
327         urlSet = urlSet.excludeJavaHome();
328         urlSet = urlSet.excludePaths(System.getProperty("sun.boot.class.path", ""));
329         urlSet = urlSet.exclude(".*/JavaVM.framework/.*");
330 
331         if (includeJars == null) {
332             urlSet = urlSet.exclude(".*?jar(!/)?");
333         } else {
334             //jar urls regexes were specified
335             List<URL> rawIncludedUrls = urlSet.getUrls();
336             Set<URL> includeUrls = new HashSet<URL>();
337             boolean[] patternUsed = new boolean[includeJars.length];
338 
339             for (URL url : rawIncludedUrls) {
340                 if ("jar".equalsIgnoreCase(url.getProtocol())) {
341                     //it is a jar file, make sure it macthes at least a url regex
342                     for (int i = 0; i < includeJars.length; i++) {
343                         String includeJar = includeJars[i];
344                         if (Pattern.matches(includeJar, url.toExternalForm())) {
345                             includeUrls.add(url);
346                             patternUsed[i] = true;
347                             break;
348                         }
349                     }
350                 } else {
351                     //it is not a jar
352                     includeUrls.add(url);
353                 }
354             }
355 
356             if (LOG.isWarnEnabled()) {
357                 for (int i = 0; i < patternUsed.length; i++) {
358                     if (!patternUsed[i]) {
359                         LOG.warn("The includeJars pattern [#0] did not match any jars in the classpath", includeJars[i]);
360                     }
361                 }
362             }
363             return new UrlSet(includeUrls);
364         }
365 
366         return urlSet;
367     }

373     protected Test<ClassFinder.ClassInfo> getPackageFinderTest(final String packageName) {
374         // so "my.package" does not match "my.package2.test"
375         final String strictPackageName = packageName + ".";
376         return new Test<ClassFinder.ClassInfo>() {
377             public boolean test(ClassFinder.ClassInfo classInfo) {
378                 String classPackageName = classInfo.getPackageName();
379                 boolean inPackage = classPackageName.equals(packageName) || classPackageName.startsWith(strictPackageName);
380                 boolean nameMatches = classInfo.getName().endsWith(actionSuffix);
381 
382                 try {
383                     return inPackage && (nameMatches || (checkImplementsAction && com.opensymphony.xwork2.Action.class.isAssignableFrom(classInfo.get())));
384                 } catch (ClassNotFoundException ex) {
385                     if (LOG.isErrorEnabled())
386                         LOG.error("Unable to load class [#0]", ex, classInfo.getName());
387                     return false;
388                 }
389             }
390         };
391     }

393     protected Test<ClassFinder.ClassInfo> getPackageLocatorTest(final String packageLocator) {
394         return new Test<ClassFinder.ClassInfo>() {
395             public boolean test(ClassFinder.ClassInfo classInfo) {
396                 String packageName = classInfo.getPackageName();
397                 if (packageName.length() > 0 && (packageLocatorsBasePackage == null || packageName.startsWith(packageLocatorsBasePackage))) {
398                     String[] splitted = packageName.split("//.");
399 
400                     boolean packageMatches = StringTools.contains(splitted, packageLocator, false);
401                     boolean nameMatches = classInfo.getName().endsWith(actionSuffix);
402 
403                     try {
404                         return packageMatches && (nameMatches || (checkImplementsAction && com.opensymphony.xwork2.Action.class.isAssignableFrom(classInfo.get())));
405                     } catch (ClassNotFoundException ex) {
406                         if (LOG.isErrorEnabled())
407                             LOG.error("Unable to load class [#0]", ex, classInfo.getName());
408                         return false;
409                     }
410                 } else
411                     return false;
412             }
413         };
414     }

416     @SuppressWarnings("unchecked")
417     protected void buildConfiguration(Set<Class> classes) {
418         Map<String, PackageConfig.Builder> packageConfigs = new HashMap<String, PackageConfig.Builder>();
419 
420         for (Class<?> actionClass : classes) {
421             // Skip all interfaces, enums, annotations, and abstract classes
422             if (actionClass.isAnnotation() || actionClass.isInterface() || actionClass.isEnum() ||
423                     (actionClass.getModifiers() & Modifier.ABSTRACT) != 0) {
424                 continue;
425             }
426 
427             // Tell the ObjectFactory about this class
428             try {
429                 objectFactory.getClassInstance(actionClass.getName());
430             } catch (ClassNotFoundException e) {
431                 // Impossible
432                 new Throwable().printStackTrace();
433                 System.exit(1);
434             }
435 
436             // Determine the action package
437             String actionPackage = actionClass.getPackage().getName();
438             if (LOG.isDebugEnabled()) {
439                 LOG.debug("Processing class [#0] in package [#1]", actionClass.getName(), actionPackage);
440             }
441 
442             // Determine the default namespace and action name
443             List<String> namespaces = determineActionNamespace(actionClass);
444             for (String namespace : namespaces) {
445                 String defaultActionName = determineActionName(actionClass);
446                 String defaultActionMethod = "execute";
447                 PackageConfig.Builder defaultPackageConfig = getPackageConfig(packageConfigs, namespace,
448                         actionPackage, actionClass, null);
449 
450                 // Verify that the annotations have no errors and also determine if the default action
451                 // configuration should still be built or not.
452                 Map<String, List<Action>> map = getActionAnnotations(actionClass);
453                 Set<String> actionNames = new HashSet<String>();
454                 if (!map.containsKey(defaultActionMethod) && ReflectionTools.containsMethod(actionClass, defaultActionMethod)) {
455                     boolean found = false;
456                     for (String method : map.keySet()) {
457                         List<Action> actions = map.get(method);
458                         for (Action action : actions) {
459 
460                             // Check if there are duplicate action names in the annotations.
461                             String actionName = action.value().equals(Action.DEFAULT_VALUE) ? defaultActionName : action.value();
462                             if (actionNames.contains(actionName)) {
463                                 throw new ConfigurationException("The action class [" + actionClass +
464                                         "] contains two methods with an action name annotation whose value " +
465                                         "is the same (they both might be empty as well).");
466                             } else {
467                                 actionNames.add(actionName);
468                             }
469 
470                             // Check this annotation is the default action
471                             if (action.value().equals(Action.DEFAULT_VALUE)) {
472                                 found = true;
473                             }
474                         }
475                     }
476 
477                     // Build the default
478                     if (!found) {
479                         createActionConfig(defaultPackageConfig, actionClass, defaultActionName, defaultActionMethod, null);
480                     }
481                 }
482 
483                 // Build the actions for the annotations
484                 for (String method : map.keySet()) {
485                     List<Action> actions = map.get(method);
486                     for (Action action : actions) {
487                         PackageConfig.Builder pkgCfg = defaultPackageConfig;
488                         if (action.value().contains("/")) {
489                             pkgCfg = getPackageConfig(packageConfigs, namespace, actionPackage,
490                                     actionClass, action);
491                         }
492 
493                         createActionConfig(pkgCfg, actionClass, defaultActionName, method, action);
494                     }
495                 }
496 
497                 // some actions will not have any @Action or a default method, like the rest actions
498                 // where the action mapper is the one that finds the right method at runtime
499                 if (map.isEmpty() && mapAllMatches) {
500                     Action actionAnnotation = actionClass.getAnnotation(Action.class);
501                     createActionConfig(defaultPackageConfig, actionClass, defaultActionName, null, actionAnnotation);
502                 }
503             }
504         }
505 
506         buildIndexActions(packageConfigs);
507 
508         // Add the new actions to the configuration
509         Set<String> packageNames = packageConfigs.keySet();
510         for (String packageName : packageNames) {
511             configuration.addPackageConfig(packageName, packageConfigs.get(packageName).build());
512         }
513     }

515     /***
516      * Determines the namespace(s) for the action based on the action class. If there is a {@link Namespace}
517      * annotation on the class (including parent classes) or on the package that the class is in, than
518      * it is used. Otherwise, the Java package name that the class is in is used in conjunction with
519      * either the <b>struts.convention.action.packages</b> or <b>struts.convention.package.locators</b>
520      * configuration values. These are used to determine which part of the Java package name should
521      * be converted into the namespace for the XWork PackageConfig.
522      *
523      * @param actionClass The action class.
524      * @return The namespace or an empty string.
525      */
526     protected List<String> determineActionNamespace(Class<?> actionClass) {
527         List<String> namespaces = new ArrayList<String>();
528 
529         // Check if there is a class or package level annotation for the namespace
530         //single namespace
531         Namespace namespaceAnnotation = AnnotationTools.findAnnotation(actionClass, Namespace.class);
532         if (namespaceAnnotation != null) {
533             if (LOG.isTraceEnabled()) {
534                 LOG.trace("Using non-default action namespace from Namespace annotation of [#0]", namespaceAnnotation.value());
535             }
536 
537             namespaces.add(namespaceAnnotation.value());
538         }
539 
540         //multiple annotations
541         Namespaces namespacesAnnotation = AnnotationTools.findAnnotation(actionClass, Namespaces.class);
542         if (namespacesAnnotation != null) {
543             if (LOG.isTraceEnabled()) {
544                 StringBuilder sb = new StringBuilder();
545                 for (Namespace namespace : namespacesAnnotation.value())
546                     sb.append(namespace.value()).append(",");
547                 sb.deleteCharAt(sb.length() - 1);
548                 LOG.trace("Using non-default action namespaces from Namespaces annotation of [#0]", sb.toString());
549             }
550 
551             for (Namespace namespace : namespacesAnnotation.value())
552                 namespaces.add(namespace.value());
553         }
554 
555         //don't use default if there are annotations
556         if (!namespaces.isEmpty())
557             return namespaces;
558 
559         String pkg = actionClass.getPackage().getName();
560         String pkgPart = null;
561         if (actionPackages != null) {
562             for (String actionPackage : actionPackages) {
563                 if (pkg.startsWith(actionPackage)) {
564                     pkgPart = actionClass.getName().substring(actionPackage.length() + 1);
565                 }
566             }
567         }
568 
569         if (pkgPart == null && packageLocators != null) {
570             for (String packageLocator : packageLocators) {
571                 int index = pkg.lastIndexOf(packageLocator);
572 
573                 // This ensures that the match is at the end, beginning or has a dot on each side of it
574                 if (index >= 0 && (index + packageLocator.length() == pkg.length() || index == 0 ||
575                         (pkg.charAt(index - 1) == '.' && pkg.charAt(index + packageLocator.length()) == '.'))) {
576                     pkgPart = actionClass.getName().substring(index + packageLocator.length() + 1);
577                 }
578             }
579         }
580 
581         if (pkgPart != null) {
582             final int indexOfDot = pkgPart.lastIndexOf('.');
583             if (indexOfDot >= 0) {
584                 String convertedNamespace = actionNameBuilder.build(pkgPart.substring(0, indexOfDot));
585                 namespaces.add("/" + convertedNamespace.replace('.', '/'));
586                 return namespaces;
587             }
588         }
589 
590         namespaces.add("");
591         return namespaces;
592     }

609     /***
610      * Locates all of the {@link Actions} and {@link Action} annotations on methods within the Action
611      * class and its parent classes.
612      *
613      * @param actionClass The action class.
614      * @return The list of annotations or an empty list if there are none.
615      */
616     protected Map<String, List<Action>> getActionAnnotations(Class<?> actionClass) {
617         Method[] methods = actionClass.getMethods();
618         Map<String, List<Action>> map = new HashMap<String, List<Action>>();
619         for (Method method : methods) {
620             Actions actionsAnnotation = method.getAnnotation(Actions.class);
621             if (actionsAnnotation != null) {
622                 Action[] actionArray = actionsAnnotation.value();
623                 boolean valuelessSeen = false;
624                 List<Action> actions = new ArrayList<Action>();
625                 for (Action ann : actionArray) {
626                     if (ann.value().equals(Action.DEFAULT_VALUE) && !valuelessSeen) {
627                         valuelessSeen = true;
628                     } else if (ann.value().equals(Action.DEFAULT_VALUE)) {
629                         throw new ConfigurationException("You may only add a single Action " +
630                                 "annotation that has no value parameter.");
631                     }
632 
633                     actions.add(ann);
634                 }
635 
636                 map.put(method.getName(), actions);
637             } else {
638                 Action ann = method.getAnnotation(Action.class);
639                 if (ann != null) {
640                     map.put(method.getName(), Arrays.asList(ann));
641                 }
642             }
643         }
644 
645         return map;
646     }

648     /***
649      * Creates a single ActionConfig object.
650      *
651      * @param pkgCfg       The package the action configuration instance will belong to.
652      * @param actionClass  The action class.
653      * @param actionName   The name of the action.
654      * @param actionMethod The method that the annotation was on (if the annotation is not null) or
655      *                     the default method (execute).
656      * @param annotation   The ActionName annotation that might override the action name and possibly
657      */
658     protected void createActionConfig(PackageConfig.Builder pkgCfg, Class<?> actionClass, String actionName,
659                                       String actionMethod, Action annotation) {
660         if (annotation != null) {
661             actionName = annotation.value() != null && annotation.value().equals(Action.DEFAULT_VALUE) ?
662                     actionName : annotation.value();
663             actionName = StringTools.lastToken(actionName, "/");
664         }
665 
666         ActionConfig.Builder actionConfig = new ActionConfig.Builder(pkgCfg.getName(),
667                 actionName, actionClass.getName());
668         actionConfig.methodName(actionMethod);
669 
670         if (LOG.isDebugEnabled()) {
671             LOG.debug("Creating action config for class [#0], name [#1] and package name [#2] in namespace [#3]",
672                     actionClass.toString(), actionName, pkgCfg.getName(), pkgCfg.getNamespace());
673         }
674 
675         //build interceptors
676         List<InterceptorMapping> interceptors = interceptorMapBuilder.build(actionClass, pkgCfg, actionName, annotation);
677         actionConfig.addInterceptors(interceptors);
678 
679         //build results
680         Map<String, ResultConfig> results = resultMapBuilder.build(actionClass, annotation, actionName, pkgCfg.build());
681         actionConfig.addResultConfigs(results);
682 
683         //add params
684         if (annotation != null)
685             actionConfig.addParams(StringTools.createParameterMap(annotation.params()));
686 
687         //add exception mappings from annotation
688         if (annotation != null && annotation.exceptionMappings() != null)
689             actionConfig.addExceptionMappings(buildExceptionMappings(annotation.exceptionMappings(), actionName));
690 
691         //add exception mapping from class
692         ExceptionMappings exceptionMappings = actionClass.getAnnotation(ExceptionMappings.class);
693         if (exceptionMappings != null)
694             actionConfig.addExceptionMappings(buildExceptionMappings(exceptionMappings.value(), actionName));
695 
696         //add
697         pkgCfg.addActionConfig(actionName, actionConfig.build());
698 
699         //check if an action with the same name exists on that package (from XML config probably)
700         PackageConfig existingPkg = configuration.getPackageConfig(pkgCfg.getName());
701         if (existingPkg != null) {
702             // there is a package already with that name, check action
703             ActionConfig existingActionConfig = existingPkg.getActionConfigs().get(actionName);
704             if (existingActionConfig != null && LOG.isWarnEnabled())
705                 LOG.warn("Duplicated action definition in package [#0] with name [#1].", pkgCfg.getName(), actionName);
706         }
707 
708         //watch class file
709         if (isReloadEnabled()) {
710             URL classFile = actionClass.getResource(actionClass.getSimpleName() + ".class");
711             FileManager.loadFile(classFile, false);
712             loadedFileUrls.add(classFile.toString());
713         }
714     }

733     private PackageConfig.Builder getPackageConfig(final Map<String, PackageConfig.Builder> packageConfigs,
734                                                    String actionNamespace, final String actionPackage, final Class<?> actionClass,
735                                                    Action action) {
736         if (action != null && !action.value().equals(Action.DEFAULT_VALUE)) {
737             if (LOG.isTraceEnabled()) {
738                 LOG.trace("Using non-default action namespace from the Action annotation of [#0]", action.value());
739             }
740             actionNamespace = StringTools.upToLastToken(action.value(), "/");
741         }
742 
743         // Next grab the parent annotation from the class
744         ParentPackage parent = AnnotationTools.findAnnotation(actionClass, ParentPackage.class);
745         String parentName = null;
746         if (parent != null) {
747             if (LOG.isTraceEnabled()) {
748                 LOG.trace("Using non-default parent package from annotation of [#0]", parent.value());
749             }
750 
751             parentName = parent.value();
752         }
753 
754         // Finally use the default
755         if (parentName == null) {
756             parentName = defaultParentPackage;
757         }
758 
759         if (parentName == null) {
760             throw new ConfigurationException("Unable to determine the parent XWork package for the action class [" +
761                     actionClass.getName() + "]");
762         }
763 
764         PackageConfig parentPkg = configuration.getPackageConfig(parentName);
765         if (parentPkg == null) {
766             throw new ConfigurationException("Unable to locate parent package [" + parentName + "]");
767         }
768 
769         // Grab based on package-namespace and if it exists, we need to ensure the existing one has
770         // the correct parent package. If not, we need to create a new package config
771         String name = actionPackage + "#" + parentPkg.getName() + "#" + actionNamespace;
772         PackageConfig.Builder pkgConfig = packageConfigs.get(name);
773         if (pkgConfig == null) {
774             pkgConfig = new PackageConfig.Builder(name).namespace(actionNamespace).addParent(parentPkg);
775             packageConfigs.put(name, pkgConfig);
776 
777             //check for @DefaultInterceptorRef in the package
778             DefaultInterceptorRef defaultInterceptorRef = AnnotationTools.findAnnotation(actionClass, DefaultInterceptorRef.class);
779             if (defaultInterceptorRef != null) {
780                 pkgConfig.defaultInterceptorRef(defaultInterceptorRef.value());
781 
782                 if (LOG.isTraceEnabled())
783                     LOG.trace("Setting [#0] as the default interceptor ref for [#1]", defaultInterceptorRef.value(), pkgConfig.getName());
784             }
785         }
786 
787         if (LOG.isTraceEnabled()) {
788             LOG.trace("Created package config named [#0] with a namespace [#1]", name, actionNamespace);
789         }
790 
791         return pkgConfig;
792     }

794     /***
795      * Determine all the index handling actions and results based on this logic:
796      *
797      * 1. Loop over all the namespaces such as /foo and see if it has an action named index
798      * 2. If an action doesn't exists in the parent namespace of the same name, create an action
799      * in the parent namespace of the same name as the namespace that points to the index
800      * action in the namespace. e.g. /foo -> /foo/index
801      * 3. Create the action in the namespace for empty string if it doesn't exist. e.g. /foo/
802      * the action is "" and the namespace is /foo
803      *
804      * @param packageConfigs Used to store the actions.
805      */
806     protected void buildIndexActions(Map<String, PackageConfig.Builder> packageConfigs) {
807         Map<String, PackageConfig.Builder> byNamespace = new HashMap<String, PackageConfig.Builder>();
808         Collection<PackageConfig.Builder> values = packageConfigs.values();
809         for (PackageConfig.Builder packageConfig : values) {
810             byNamespace.put(packageConfig.getNamespace(), packageConfig);
811         }
812 
813         // Step #1
814         Set<String> namespaces = byNamespace.keySet();
815         for (String namespace : namespaces) {
816             // First see if the namespace has an index action
817             PackageConfig.Builder pkgConfig = byNamespace.get(namespace);
818             ActionConfig indexActionConfig = pkgConfig.build().getAllActionConfigs().get("index");
819             if (indexActionConfig == null) {
820                 continue;
821             }
822 
823             // Step #2
824             if (!redirectToSlash) {
825                 int lastSlash = namespace.lastIndexOf('/');
826                 if (lastSlash >= 0) {
827                     String parentAction = namespace.substring(lastSlash + 1);
828                     String parentNamespace = namespace.substring(0, lastSlash);
829                     PackageConfig.Builder parent = byNamespace.get(parentNamespace);
830                     if (parent == null || parent.build().getAllActionConfigs().get(parentAction) == null) {
831                         if (parent == null) {
832                             parent = new PackageConfig.Builder(parentNamespace).namespace(parentNamespace).
833                                     addParents(pkgConfig.build().getParents());
834                             packageConfigs.put(parentNamespace, parent);
835                         }
836 
837                         if (parent.build().getAllActionConfigs().get(parentAction) == null) {
838                             parent.addActionConfig(parentAction, indexActionConfig);
839                         }
840                     } else if (LOG.isTraceEnabled()) {
841                         LOG.trace("The parent namespace [#0] already contains " +
842                                 "an action [#1]", parentNamespace, parentAction);
843                     }
844                 }
845             }
846 
847             // Step #3
848             if (pkgConfig.build().getAllActionConfigs().get("") == null) {
849                 if (LOG.isTraceEnabled()) {
850                     LOG.trace("Creating index ActionConfig with an action name of [] for the action " +
851                             "class [#0]", indexActionConfig.getClassName());
852                 }
853 
854                 pkgConfig.addActionConfig("", indexActionConfig);
855             }
856         }
857     }
