526     private PackageConfig makePackageConfig(String name, String namespace, PackageConfig parent,
527             String defaultResultType, ResultTypeConfig[] results, List<InterceptorConfig> interceptors,
528             List<InterceptorStackConfig> interceptorStacks) {
529         PackageConfig.Builder builder = new PackageConfig.Builder(name);
530         if (namespace != null) {
531             builder.namespace(namespace);
532         }
533         if (parent != null) {
534             builder.addParent(parent);
535         }
536         if (defaultResultType != null) {
537             builder.defaultResultType(defaultResultType);
538         }
539         if (results != null) {
540             for (ResultTypeConfig result : results) {
541                 builder.addResultTypeConfig(result);
542             }
543         }
544         if (interceptors != null) {
545             for (InterceptorConfig ref : interceptors) {
546                 builder.addInterceptorConfig(ref);
547             }
548         }
549         if (interceptorStacks != null) {
550             for (InterceptorStackConfig ref : interceptorStacks) {
551                 builder.addInterceptorStackConfig(ref);
552             }
553         }
554 
555         return new MyPackageConfig(builder.build());
